= .postcssrc.js
* Node.js 환경에서는 CommmonJs 방식을 사용하기 때문에
* import 를 사용할수 없어  require() 를 대신 사용한다.
* export 를 사용할수 없어  module.exports 를 대신 사용한다.
module.exports = {
    plugins: [
        require('autoprefixer')
    ]
}
